      % ---- Panel C ----
      \panelhead{Useful transfer theorem}

      {\Large \textbf{Direct:} grid-search at the target width. \textbf{Transfer:} grid-search at small $n$, train once at target. Which wins?}

      \vspace{0.7em}
      \begin{center}
        \bluebox{\parbox{0.96\linewidth}{\centering \Large \textbf{Theorem.} Suppose the gaps decay as power laws, $a_n \sim n^{-\alpha}$ and $b_n \sim n^{-\beta}$, near a strongly-convex minimum of the infinite-width loss. Then
        \[ \text{\textbf{transfer} beats \textbf{direct} asymptotically}\quad\iff\quad \beta > \tfrac{\alpha}{2}, \quad\text{i.e.,}\quad b_n = o\!\bigl(\sqrt{a_n}\bigr). \]
        \vspace{-0.8em}
        {\normalsize\upshape (same condition as \emph{fast transfer}).}}}
      \end{center}

      \vspace{0.5em}

      {\Large \textbf{Why:} transfer pays $c_n \propto b_n^2$ (HP error); direct pays $\sim a_n$ (loss gap). Transfer wins iff $b_n^2 = o(a_n)$.}

      \vspace{0.2em}
      \begin{center}{\large\itshape Two synthetic testbeds: 2-layer ReLU learning a ball-indicator target (left); random-feature ridge regression in $d$ dimensions (right).}\end{center}

      \vspace{0.2em}
      \begin{columns}[c, onlytextwidth]

          \includegraphics[width=\linewidth, height=9cm, keepaspectratio]{slow.png}\\[0.3em]
          {\large\itshape \textbf{$\mu$P failure} --- HP gap decays like $b_n \propto n^{-0.39}\!\approx\!\sqrt{a_n}$. Only the baseline; transfer breaks even.}
        \end{column}
        \begin{column}{0.41\linewidth}\centering

          {\large\itshape \textbf{Provable fast transfer} --- both $a_n, b_n\!\sim\!d/n$, so $b_n^2\!=\!o(a_n)$. Theorem's condition holds.}
        \end{column}

      % ---- Panel C ----
      \panelhead{The mechanism: low-rank trajectory decomposition}

      {\Large Intuition: optimal HPs should depend on network statistics that stabilize with width \emph{faster} than the loss itself. We decompose the stepwise linearized loss change via the eigenvalues of a gradient/update alignment matrix:}

      \vspace{0.4em}
      \begin{columns}[c, onlytextwidth]
        \begin{column}{0.58\linewidth}\centering
          \includegraphics[width=\linewidth, height=18cm, keepaspectratio]{decomposition_equation.pdf}
        \end{column}
        \begin{column}{0.42\linewidth}\centering
          \includegraphics[width=\linewidth, height=14cm, keepaspectratio]{decomp.png}\\[0.5em]
          {\large\itshape Cartoon: total loss $=$ (U-shaped, width-stable) top-$k$ $+$ (flat) residual.}
        \end{column}
      \end{columns}

      \vspace{0.4em}

      \begin{center}
        \tealbox{\parbox{0.96\linewidth}{\normalsize \textbf{Conjecture.} For some $k$: \textbf{(1)} the top-$k$ loss is locally strongly convex at its optimum; \textbf{(2)} \emph{top-$k$ invariance} --- the top-$k$ loss and its optimal HP are approximately width-independent; \textbf{(3)} \emph{residual flatness} --- the residual loss is flat in HP, so it cannot move the optimum.}}
      \end{center}

      \vspace{0.7em}
